\begin{frame}{Estimaciones de MCO}
\protect\hypertarget{estimaciones-de-mco}{}
Modelo econométrico:

\[
y_i = \beta_0 + \beta_1x_i + u_i
\]

\begin{block}{Ecuación:}
\protect\hypertarget{ecuaciuxf3n}{}
Variables: \(y\) dependiente y \(x\) independiente.

Parametros:

\(\beta_0\) intercepto o constante.

\(\beta_1\) es la pendiente de la recta, es la relación entre \(y\) y
\(x\).

\(u\) es el término de error, representa los factores distintos a \(x\)
que afectan a \(y\) .
\end{block}
\end{frame}
